            cols.append(col)



    export_df = export_df[cols]



    with pd.ExcelWriter(output, engine="openpyxl") as writer:

def floor_2(value):

    return math.floor(float(value) * 100) / 100





def normalize_col_name(name):

    return re.sub(r"[\s_\-\[\]\(\)]", "", str(name)).lower()





def find_column(df, candidates):

    norm_map = {normalize_col_name(col): col for col in df.columns}

    for candidate in candidates:

        matched = norm_map.get(normalize_col_name(candidate))

        if matched:

            return matched

    return None





def has_value(value):

    if value is None:

        return False

    if isinstance(value, float) and pd.isna(value):

        return False

    return str(value).strip() != ""





def coerce_amount_series(series):

    """콤마/문자가 섞인 금액 문자열을 float 숫자로 강제 변환."""

    cleaned = (

        series.astype(str)

        .str.replace(",", "", regex=False)

        .str.replace(r"[^\d.\-]", "", regex=True)

    )

    return pd.to_numeric(cleaned, errors="coerce").fillna(0.0)





def won_to_eok(amount_won):

    return float(amount_won) / WON_PER_EOK





def to_number(value):

    if value is None or (isinstance(value, float) and pd.isna(value)):

        return 0.0

    if isinstance(value, (int, float)):

        return float(value)

    text = str(value).strip()

    if not text:

        return 0.0

    cleaned = re.sub(r"[,\s원건회억]", "", text)

    match = re.search(r"-?\d+(?:\.\d+)?", cleaned)

    if not match:

        return 0.0

    return float(match.group())





def to_datetime(value):

    if not has_value(value):

        return pd.NaT

    return pd.to_datetime(value, errors="coerce")





def parse_pipe_numbers(value):

    if not has_value(value):

        return []

    text = str(value).replace("/", "|").replace(",", "|")

    return [to_number(piece) for piece in text.split("|") if has_value(piece)]





def classify_orderer(orderer_text):

    text = str(orderer_text or "").strip()

    if not text:

        return "기타"

    if any(keyword in text for keyword in ["국토부", "환경부", "청", "사무소"]):

        return "국가기관"

    if any(keyword in text for keyword in ["특별시", "광역시", "자치시", "시청", "도청", "군청", "구청", "사업소"]):

        return "지방자치단체"

    if text.endswith(("시", "도", "군", "구")):

        return "지방자치단체"

    if any(keyword in text for keyword in ["공사", "공단"]):

        return "공공기관"

    return "기타"





def get_base_category(text):

    """괄호 앞 텍스트만 추출하는 결정론적 매칭 (예: 하천(기본계획) -> 하천)."""

    return str(text).split("(")[0].strip()





def find_date_column_indices_from_header_rows(raw_df, max_scan_rows=3):

    """상위 N개 행에서 '시작일'/'종료일' 텍스트가 있는 열 인덱스를 탐색."""

    start_idx = None

    end_idx = None

    for row_i in range(min(max_scan_rows, len(raw_df))):

        for col_i, val in enumerate(raw_df.iloc[row_i].tolist()):

            norm = str(val).replace(" ", "")

            if not norm:

                continue

            if start_idx is None and ("시작일" in norm or "착수일" in norm):

                start_idx = col_i

            if end_idx is None and ("종료일" in norm or "준공일" in norm):

                end_idx = col_i

    return start_idx, end_idx





def apply_date_column_rename(df, start_idx, end_idx):

    """열 인덱스 기반으로 시작일/종료일 컬럼명을 강제 지정."""

    cols = list(df.columns)

    rename_map = {}

    if start_idx is not None and start_idx < len(cols):

        rename_map[cols[start_idx]] = "시작일"

    if end_idx is not None and end_idx < len(cols) and end_idx != start_idx:

        rename_map[cols[end_idx]] = "종료일"

    if rename_map:

        df = df.rename(columns=rename_map)

    return df





def fallback_rename_date_pattern_columns(df):

    """헤더 탐색 실패 시 데이터 패턴 기반으로 시작일/종료일 추정."""

    normalized_cols = [normalize_col_name(col) for col in df.columns]

    has_start_col = any("시작일" in col or "착수일" in col for col in normalized_cols)

    has_end_col = any("종료일" in col or "준공일" in col for col in normalized_cols)

    if has_start_col and has_end_col:

        return df



    date_like_cols = []

    date_pattern = re.compile(r"^\d{4}[-./]\d{1,2}[-./]\d{1,2}$")

    for col in df.columns:

        series = df[col].astype(str).str.strip()

        sample = series[series != ""].head(30)

        if sample.empty:

            continue

        match_ratio = sample.apply(lambda x: bool(date_pattern.match(x))).mean()

        if match_ratio >= 0.5:

            date_like_cols.append(col)



    if len(date_like_cols) >= 2:

        rename_map = {}

        if not has_start_col:

            rename_map[date_like_cols[0]] = "시작일"

        if not has_end_col:

            second_col = date_like_cols[1] if date_like_cols[1] != date_like_cols[0] else date_like_cols[-1]

            rename_map[second_col] = "종료일"

        df = df.rename(columns=rename_map)

    return df





def extract_dataframe(raw_data, patch_multi_header_dates=False):

    raw_df = pd.DataFrame(raw_data)

    if raw_df.empty:

        return raw_df



    start_idx, end_idx = (None, None)

    if patch_multi_header_dates:

        start_idx, end_idx = find_date_column_indices_from_header_rows(raw_df, max_scan_rows=3)



    header_idx = 0

    for i, row in raw_df.iterrows():

        row_text = "".join(str(val).replace(" ", "") for val in row.values)

        if any(keyword in row_text for keyword in ["공종", "발주처", "지자체"]):

            header_idx = i

            break



    df = raw_df.iloc[header_idx + 1 :].copy()

    df.columns = [str(col).strip() for col in raw_df.iloc[header_idx].tolist()]

    df = df.reset_index(drop=True)



    if patch_multi_header_dates:

        df = apply_date_column_rename(df, start_idx, end_idx)

    df = fallback_rename_date_pattern_columns(df)

    return df

def prepare_company_dataframe(company_df):

    company = company_df.copy()

    orderer_col = find_column(company, ["발주처", "발주기관", "기관명"])

    biz_col = find_column(company, ["사업구분", "용역구분", "설계/감리", "구분"])

    work_type_col = find_column(company, ["공종", "분야", "업종"])

    total_amount_col = find_column(company, ["준공금액", "총금액", "전체금액", "계약금액", "금액", "실적금액"])

    share_amount_col = find_column(company, ["지분금액", "지분", "참여금액"])

    count_col = find_column(company, ["건수", "실적건수"])


    project_col = find_column(company, ["사업명", "용역명", "공사명", "프로젝트명"])

    overview_col = find_column(company, ["사업개요", "개요"])

    remarks_col = find_column(company, ["비고"])

    cems_col = find_column(company, ["CEMS", "cems"])

    certificate_col = find_column(company, ["실적증명서 발급여부", "실적증명서발급여부"])



    if orderer_col is None or work_type_col is None:

        raise ValueError("회사실적 시트에서 '발주처' 또는 '공종' 컬럼을 찾지 못했습니다.")



    company["실적ID"] = range(1, len(company) + 1)

    company["발주처_유형"] = company[orderer_col].apply(classify_orderer)

    company["발주처"] = company[orderer_col].astype(str).str.strip()

    company["사업구분"] = company[biz_col].astype(str).str.strip() if biz_col else ""

    company["공종"] = company[work_type_col].astype(str).str.strip()

    company["공종_원본"] = company["공종"]

    company["사업명"] = (

        company[project_col].astype(str).str.strip() if project_col else company["공종"]

    )

    if overview_col:

        company["사업개요"] = company[overview_col].astype(str).str.strip()

    else:

        company["사업개요"] = ""

    if remarks_col:

        company["비고"] = company[remarks_col].astype(str).str.strip()

    else:

        company["비고"] = ""

    if cems_col:

        company["CEMS"] = company[cems_col].astype(str).str.strip()

    else:

        company["CEMS"] = ""

    if total_amount_col:

        company["준공금액"] = coerce_amount_series(company[total_amount_col])

    else:

        company["준공금액"] = 0.0

    if share_amount_col:

        company["지분금액"] = coerce_amount_series(company[share_amount_col])

    else:

        company["지분금액"] = company["준공금액"]

    company["건수"] = pd.to_numeric(company[count_col].apply(to_number), errors="coerce").fillna(0.0) if count_col else 1.0

    # [1] 백엔드 연산용 (시작일/종료일): 반드시 datetime으로 파싱 (기존 로직 유지, 에러 없음)
    calc_start_col = find_column(company, ["시작일"])
    calc_end_col = find_column(company, ["종료일"])
    company["시작일"] = company[calc_start_col].apply(to_datetime) if calc_start_col else pd.NaT
    company["종료일"] = company[calc_end_col].apply(to_datetime) if calc_end_col else pd.NaT

    # [2] 프론트엔드 표출용 (착수일/준공일): 줄바꿈(\n)을 " / "로 치환해 단일 셀 표출
    disp_start_col = find_column(company, ["착수일"])
    disp_end_col = find_column(company, ["준공일"])
    company["착수일"] = (
        company[disp_start_col]
        .astype(str)
        .str.strip()
        .replace(["nan", "None", "NaT"], "")
        .str.replace(r"\r?\n", " / ", regex=True)
        if disp_start_col
        else ""
    )
    company["준공일"] = (
        company[disp_end_col]
        .astype(str)
        .str.strip()
        .replace(["nan", "None", "NaT"], "")
        .str.replace(r"\r?\n", " / ", regex=True)
        if disp_end_col
        else ""
    )

    # [3] Vectorized Fallback: 표출용 텍스트가 비어있으면, 연산용 날짜를 YYYY-MM-DD 문자열로 변환하여 채움 (빈 값 에러 원천 차단)
    fmt_start = company["시작일"].dt.strftime("%Y-%m-%d").fillna("")
    fmt_end = company["종료일"].dt.strftime("%Y-%m-%d").fillna("")
    company["착수일"] = company["착수일"].where(company["착수일"] != "", fmt_start)
    company["준공일"] = company["준공일"].where(company["준공일"] != "", fmt_end)
    # 표출용 데이터: 대시를 점으로 치환하여 YYYY.MM.DD 형식 강제 고정
    company["착수일"] = company["착수일"].astype(str).str.replace("-", ".", regex=False)
    company["준공일"] = company["준공일"].astype(str).str.replace("-", ".", regex=False)

    company["선택"] = True

    company["선택"] = company["선택"].astype(bool)

    company["인정률(%)"] = 100.0

    if certificate_col:

        company["실적증명서 발급여부"] = company[certificate_col].astype(str).str.strip()

    else:

        company["실적증명서 발급여부"] = ""



    return company[company["발주처_유형"] != "기타"].copy()

def filter_criteria_master_by_business_type(criteria_df, columns, business_type):

    """역추적 검색용 — 전체 마스터에서 현재 사업구분(설계/감리)만 1차 추출."""

    criteria = criteria_df.copy()

    biz_col = columns["biz_type"]

    if biz_col:

        criteria[biz_col] = criteria[biz_col].astype(str).str.strip()

        return criteria[criteria[biz_col] == business_type].copy()



    if business_type == "설계":

        masks = []

        if columns["design_count_hurdle"]:

            masks.append(criteria[columns["design_count_hurdle"]].apply(has_value))

        if columns["design_amount_hurdle"]:

            masks.append(criteria[columns["design_amount_hurdle"]].apply(has_value))

        if masks:

            combined = masks[0]

            for mask in masks[1:]:

                combined = combined | mask

            return criteria[combined].copy()

        return criteria.iloc[0:0].copy()



    if columns["supervision_hurdle"]:

        return criteria[criteria[columns["supervision_hurdle"]].apply(has_value)].copy()

    return criteria.iloc[0:0].copy()





def highlight_and_preserve(text, criteria_keyword):

    """원문 전체 표출 + 키워드 시각적 앵커 (【 ★ 키워드 ★ 】)."""

    if pd.isna(text):

        return text



    keyword = str(criteria_keyword or "").strip()

    if not keyword:

        return str(text)



    text_str = str(text)

    pattern = re.compile(f"({re.escape(keyword)})", re.IGNORECASE)

    highlight_replacement = r"【 ★ \1 ★ 】"

def filter_criteria_for_target(criteria_df, columns, target_owner, business_type):

    criteria = criteria_df.copy()

    owner_col = columns["owner"]

    work_type_col = columns["work_type"]



    if owner_col is None or work_type_col is None:

        raise ValueError("세부평가기준 시트에서 '발주처' 또는 '공종' 컬럼을 찾지 못했습니다.")



    criteria[owner_col] = criteria[owner_col].astype(str).str.strip()

    criteria[work_type_col] = criteria[work_type_col].astype(str).str.strip()

    criteria = criteria[criteria[owner_col] == target_owner].copy()

    criteria = criteria[criteria[work_type_col] != ""].copy()



    biz_col = columns["biz_type"]

    if biz_col:

        criteria[biz_col] = criteria[biz_col].astype(str).str.strip()

        criteria = criteria[criteria[biz_col] == business_type]

    elif business_type == "설계":

        masks = []

        if columns["design_count_hurdle"]:

            masks.append(criteria[columns["design_count_hurdle"]].apply(has_value))

        if columns["design_amount_hurdle"]:

            masks.append(criteria[columns["design_amount_hurdle"]].apply(has_value))

        if masks:

            combined = masks[0]

            for mask in masks[1:]:

                combined = combined | mask

            criteria = criteria[combined].copy()

        else:

            criteria = criteria.iloc[0:0].copy()

    else:

        if columns["supervision_hurdle"]:

            criteria = criteria[criteria[columns["supervision_hurdle"]].apply(has_value)].copy()

        else:

            criteria = criteria.iloc[0:0].copy()



    return criteria





def recognition_window_days(recognition_years):

    """윤년 배제: N년 = N * 365일 절대 산술."""

    return int(recognition_years) * 365





def filter_company_for_work_type(

    company_df,

    selected_work_type,

    business_type,

    base_date,

    recognition_years,

    amount_filter_col,

    min_amount_eok,

):

    """

    데이터 격리 절대 락(Lock):

    1) 사업구분 == selected_biz_type

    2) 공종 == get_base_category(selected_work_type)

    3) 종료일이 공고일 - (N*365일) 이전이면 제외 (종료일 NaT 진행중 건은 유지)

    4) 선택 금액 컬럼이 (N억 * 1억원) 미만이면 제외

    """

    match_base = get_base_category(selected_work_type)

    lock_mask = (company_df["사업구분"].astype(str).str.strip() == str(business_type).strip()) & (

        company_df["공종"].astype(str).str.strip() == match_base

    )

    filtered = company_df.loc[lock_mask].copy()



    window_days = recognition_window_days(recognition_years)

    cutoff_date = pd.Timestamp(base_date) - timedelta(days=window_days)

    end_dates = pd.to_datetime(filtered["종료일"], errors="coerce")

    ongoing_mask = end_dates.isna()

    within_period_mask = end_dates >= cutoff_date

    period_mask = ongoing_mask | within_period_mask



    min_amount_won = float(min_amount_eok) * WON_PER_EOK

    amount_values = pd.to_numeric(filtered[amount_filter_col], errors="coerce").fillna(0.0)

    amount_mask = amount_values >= min_amount_won



    return filtered.loc[period_mask & amount_mask].copy()

def pad_lists(*lists):

    max_len = max((len(lst) for lst in lists if lst), default=0)

    if max_len == 0:

        return lists

    padded = []

    for lst in lists:

        if not lst:

            padded.append([""] * max_len)

        elif len(lst) < max_len:

            padded.append(lst + [""] * (max_len - len(lst)))

        else:

            padded.append(lst[:max_len])

    return padded





def build_design_breakdown_table(criteria_row, columns):

    count_hurdles = parse_pipe_numbers(criteria_row.get(columns["design_count_hurdle"], ""))

    count_scores = parse_pipe_numbers(criteria_row.get(columns["design_count_score"], ""))

    amount_hurdles = parse_pipe_numbers(criteria_row.get(columns["design_amount_hurdle"], ""))

    amount_scores = parse_pipe_numbers(criteria_row.get(columns["design_amount_score"], ""))



    count_hurdles, count_scores, amount_hurdles, amount_scores = pad_lists(

        count_hurdles, count_scores, amount_hurdles, amount_scores

    )

    max_len = max(len(count_hurdles), len(amount_hurdles), 1)



    rows = []

    for i in range(max_len):

        rows.append(

            {

                "건수": count_hurdles[i] if i < len(count_hurdles) else "",

                "건수배점": count_scores[i] if i < len(count_scores) else "",

                "금액": amount_hurdles[i] if i < len(amount_hurdles) else "",

                "금액배점": amount_scores[i] if i < len(amount_scores) else "",

            }

        )

    return pd.DataFrame(rows)





def build_supervision_breakdown_table(criteria_row, columns):

    hurdles = parse_pipe_numbers(criteria_row.get(columns["supervision_hurdle"], ""))

    scores = parse_pipe_numbers(criteria_row.get(columns["supervision_score"], ""))

    hurdles, scores = pad_lists(hurdles, scores)

    max_len = max(len(hurdles), 1)



    rows = []

    for i in range(max_len):

        rows.append(

            {

                "건수": "",

                "건수배점": "",

                "금액": hurdles[i] if i < len(hurdles) else "",

                "금액배점": scores[i] if i < len(scores) else "",

            }

        )

    return pd.DataFrame(rows)





def evaluate_score(actual_value, hurdle_value, score_value):

    hurdles = parse_pipe_numbers(hurdle_value)

    scores = parse_pipe_numbers(score_value)

    if not hurdles or not scores:

        return 0.0



    pairs = sorted(zip(hurdles, scores), key=lambda x: x[0], reverse=True)

    if actual_value <= 0:

        return 0.0



    for hurdle, score in pairs:

        if actual_value >= hurdle:

            return float(score)

    return 0.0





def compute_overlap_days(start_date, end_date, base_date, recognition_years):

    if pd.isna(start_date) or pd.isna(end_date):

        return 0



    start_date = pd.Timestamp(start_date)

    end_date = pd.Timestamp(end_date)

    base_date = pd.Timestamp(base_date)



    if end_date < start_date:

        start_date, end_date = end_date, start_date



    window_days = recognition_window_days(recognition_years)

    window_start = base_date - timedelta(days=window_days)

    overlap_start = max(start_date, window_start)

    overlap_end = min(end_date, base_date)



    if overlap_end < overlap_start:

        return 0

    return int((overlap_end - overlap_start).days + 1)

def apply_partial_editor_updates(*edited_slices):

    """선택/인정률(%)만 실적ID(str) 기준으로 session editor_df에 부분 갱신."""

    if not edited_slices:

        return



    base = st.session_state["editor_df"]

    base["실적ID"] = base["실적ID"].astype(str)

    base_indexed = base.set_index("실적ID")



    for edited in edited_slices:

        slice_indexed = normalize_edited_slice(edited)

        if slice_indexed.empty:

            continue

        update_cols = [col for col in EDITABLE_MERGE_COLS if col in slice_indexed.columns]

        if not update_cols:

            continue

        patch = slice_indexed[update_cols].copy()

        if "선택" in patch.columns:

            patch["선택"] = patch["선택"].fillna(False).astype(bool)

        if "인정률(%)" in patch.columns:

            patch["인정률(%)"] = pd.to_numeric(patch["인정률(%)"], errors="coerce")

        base_indexed.update(patch)



    refreshed = base_indexed.reset_index().set_index("실적ID")

    session_df = st.session_state["editor_df"]

    session_df["실적ID"] = session_df["실적ID"].astype(str)

    for col in EDITABLE_MERGE_COLS:

        if col in refreshed.columns:

            session_df[col] = session_df["실적ID"].map(refreshed[col])

def resolve_preserved_editor_source(state_key):

    """등록 상태 복원용 — 동일 화면이면 세션 editor_df 우선, 없으면 portfolio."""

    if st.session_state.get("current_state_key") == state_key and "editor_df" in st.session_state:

        return st.session_state["editor_df"].copy()



    portfolio = st.session_state["portfolio_memory"]

    if state_key in portfolio:

        return portfolio[state_key].copy()



    if st.session_state.get("force_sync") and "editor_df" in st.session_state:

        return st.session_state["editor_df"].copy()



    return None





def append_filter_excluded_registrations(visible_df, saved_df):

    """필터 밖으로 나간 등록 실적을 세션에서 유지(메모리 증발 방지)."""

    if saved_df is None or saved_df.empty:

        return visible_df



    visible = ensure_editor_bool_columns(visible_df.copy())

    saved = ensure_editor_bool_columns(saved_df.copy())

    visible["실적ID"] = visible["실적ID"].astype(str)

    saved["실적ID"] = saved["실적ID"].astype(str)



    visible_ids = set(visible["실적ID"].tolist())

    missing_saved = saved[~saved["실적ID"].isin(visible_ids)].copy()

    missing_saved = missing_saved[missing_saved["등록여부"] == True]

    if missing_saved.empty:

        return visible



    return pd.concat([visible, missing_saved], ignore_index=True)





def sync_editor_state(state_key, company_filtered):

    """최신 시트 뼈대 + 보존된 등록 상태 병합. portfolio는 💾 저장 시에만 갱신."""

    fresh_df = build_fresh_sync_frame(company_filtered)

    saved_df = resolve_preserved_editor_source(state_key)



    if saved_df is not None:

        merged_df = merge_preserved_portfolio_state(fresh_df, saved_df)

        merged_df = append_filter_excluded_registrations(merged_df, saved_df)

    else:

        merged_df = fresh_df



    st.session_state["editor_df"] = ensure_editor_bool_columns(merged_df)

    return st.session_state["editor_df"]

def format_criteria_detail_markdown(raw_text):

    """세부평가기준 원문을 마크다운 가독성 형태로 정돈."""

    text = str(raw_text or "").replace("\r\n", "\n").replace("\r", "\n").strip()

    if not text:

        return ""



    # 문장 중간에 등장하는 기호 앞에 강제 단락 분리

    text = re.sub(r"(?<!\n)\s*※", "\n\n※", text)

    text = re.sub(r"(?<!\n)\s*◦", "\n\n◦", text)

    text = re.sub(r"(?<!\n)\s*•", "\n\n•", text)



    bullet_prefixes = ("◦", "•", "-", "–", "—")

    formatted_lines = []

    for line in text.split("\n"):

        stripped = line.strip()

        if not stripped:

            formatted_lines.append("")

            continue



        if stripped.startswith("※"):

            body = stripped.lstrip("※").strip()

            formatted_lines.append(f"**※ {body}**" if body else "**※**")

        elif stripped.startswith(bullet_prefixes):

            symbol = stripped[0]

            body = stripped[1:].strip()

            formatted_lines.append(f"- {body}")

        else:

            formatted_lines.append(stripped)



    blocks = []

    current = []

    for line in formatted_lines:

        if line == "":

            if current:

                blocks.append("\n".join(current))

                current = []

        else:

            current.append(line)

    if current:

        blocks.append("\n".join(current))



    return "\n\n".join(blocks)





def render_criteria_detail(raw_text):

        return

    with st.container(border=True):

        st.markdown(markdown_text)





def filter_aggregate_source_df(edited_df, business_type):

    """Table A(등록) 또는 선택 행만 집계 대상으로 필터."""

    df = cast_numeric_columns(edited_df.copy())

    if business_type == "설계":

        if "등록여부" in df.columns:

            df["등록여부"] = df["등록여부"].fillna(False).astype(bool)

            df = df[df["등록여부"] == True].copy()

    elif "선택" in df.columns:

        df["선택"] = df["선택"].fillna(True).astype(bool)

        df = df[df["선택"] == True].copy()

    return df





def is_certificate_verified(df):

    if CERTIFICATE_STATUS_COL not in df.columns:

        return pd.Series(False, index=df.index)

    return df[CERTIFICATE_STATUS_COL].astype(str).str.upper().str.strip() == "O"





def compute_design_recognition_totals(df):

    if df.empty:

        return 0.0, 0.0

    rate = df["인정률(%)"] / 100.0

    total_count = float((df["건수"] * rate).sum())

    total_amount_won = float((df["지분금액"] * rate).sum())

    return total_count, total_amount_won





def compute_supervision_recognition_total(df, base_date, recognition_years):

    if df.empty:

        return 0.0



    work = cast_numeric_columns(df.copy())

    if "시작일" in work.columns:

        work["시작일"] = pd.to_datetime(work["시작일"], errors="coerce")

    if "종료일" in work.columns:

        work["종료일"] = pd.to_datetime(work["종료일"], errors="coerce")



    window_days = recognition_window_days(recognition_years)

    recognized_amounts = []

    for _, row in work.iterrows():

        overlap_days = compute_overlap_days(

            row["시작일"], row["종료일"], base_date, recognition_years

        )

        share_amount = float(row["지분금액"])

        rate = float(row["인정률(%)"]) / 100.0

        recognized = (overlap_days / window_days) * share_amount * rate

        recognized_amounts.append(recognized)



    return float(sum(recognized_amounts))





def aggregate_recognition_totals(edited_df, business_type, base_date, recognition_years):

    """(확정건수, 확정금액, 전체건수, 전체금액) 반환 — 확정은 발급여부 'O'만."""

    df = filter_aggregate_source_df(edited_df, business_type)

    verified_df = df[is_certificate_verified(df)].copy()



    if business_type == "설계":

        all_count, all_amount = compute_design_recognition_totals(df)

        verified_count, verified_amount = compute_design_recognition_totals(verified_df)

        return verified_count, verified_amount, all_count, all_amount



    all_amount = compute_supervision_recognition_total(df, base_date, recognition_years)

    verified_amount = compute_supervision_recognition_total(verified_df, base_date, recognition_years)

    return 0.0, verified_amount, 0.0, all_amount
